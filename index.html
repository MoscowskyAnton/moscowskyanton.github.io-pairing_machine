import asyncio import pyodide async def main(): await pyodide.loadPackage("micropip") import micropip await micropip.install("matplotlib") import matplotlib.pyplot as plt import numpy as np x = np.random.rand(100) y = np.random.rand(100) plt.scatter(x, y) plt.show() asyncio.ensure_future(main())
